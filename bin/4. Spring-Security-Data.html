1. What is Spring Security, and how does it work?

Answer:

Spring Security is a powerful framework providing authentication, authorization, and protection against common attacks for Spring applications. It integrates seamlessly with Spring Boot and offers comprehensive security features.

Basic Security Configuration (Spring Security 6.x)

@Configuration
@EnableWebSecurity
public class SecurityConfig {
    
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        return http
            .csrf(csrf -> csrf.disable())  // Disable for REST APIs
            .sessionManagement(session -> 
                session.sessionCreationPolicy(SessionCreationPolicy.STATELESS))
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/public/**", "/auth/**").permitAll()
                .requestMatchers("/admin/**").hasRole("ADMIN")
                .requestMatchers("/api/**").authenticated()
                .anyRequest().authenticated()
            )
            .httpBasic(Customizer.withDefaults())
            .build();
    }
    
    @Bean
    public PasswordEncoder passwordEncoder() {
        return new BCryptPasswordEncoder(12);
    }
}

2. How do Authentication and Authorization differ?

Answer:

Authentication vs Authorization Comparison

Aspect Authentication (AuthN) Authorization (AuthZ)
Question "Who are you?" "What can you do?"
Purpose Verify identity Control access to resources
When Happens first (login) Happens after authentication
Failure 401 Unauthorized 403 Forbidden
Example Login with username/password Admin can delete; User can only read

Complete Security Configuration Example

@Configuration
@EnableWebSecurity
@EnableMethodSecurity  // Enables @PreAuthorize, @PostAuthorize
public class SecurityConfig {
    
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        return http
            .csrf(csrf -> csrf.disable())
            .authorizeHttpRequests(auth -> auth
                // Public endpoints
                .requestMatchers("/public/**", "/health").permitAll()
                // Role-based authorization
                .requestMatchers("/admin/**").hasRole("ADMIN")
                .requestMatchers("/manager/**").hasAnyRole("ADMIN", "MANAGER")
                // Authority-based authorization  
                .requestMatchers(HttpMethod.DELETE, "/api/**").hasAuthority("DELETE_PRIVILEGE")
                // All other requests require authentication
                .anyRequest().authenticated()
            )
            .formLogin(form -> form
                .loginPage("/login")
                .defaultSuccessUrl("/dashboard")
                .permitAll()
            )
            .logout(logout -> logout
                .logoutUrl("/logout")
                .logoutSuccessUrl("/login?logout")
                .deleteCookies("JSESSIONID")
            )
            .build();
    }
    
    @Bean
    public UserDetailsService userDetailsService() {
        UserDetails user = User.builder()
            .username("user")
            .password(passwordEncoder().encode("password"))
            .roles("USER")
            .build();
        
        UserDetails admin = User.builder()
            .username("admin")
            .password(passwordEncoder().encode("admin123"))
            .roles("ADMIN", "USER")
            .authorities("DELETE_PRIVILEGE", "ROLE_ADMIN", "ROLE_USER")
            .build();
        
        return new InMemoryUserDetailsManager(user, admin);
    }
    
    @Bean
    public PasswordEncoder passwordEncoder() {
        return new BCryptPasswordEncoder();
    }
}

JWT Implementation

@Component
public class JwtTokenProvider {
    
    @Value("${jwt.secret}")
    private String secretKey;
    
    @Value("${jwt.expiration:3600000}")  // 1 hour default
    private long validityInMs;
    
    private SecretKey key;
    
    @PostConstruct
    protected void init() {
        this.key = Keys.hmacShaKeyFor(secretKey.getBytes(StandardCharsets.UTF_8));
    }
    
    public String createToken(String username, List<String> roles) {
        Date now = new Date();
        Date validity = new Date(now.getTime() + validityInMs);
        
        return Jwts.builder()
            .subject(username)
            .claim("roles", roles)
            .issuedAt(now)
            .expiration(validity)
            .signWith(key)
            .compact();
    }
    
    public boolean validateToken(String token) {
        try {
            Jwts.parser().verifyWith(key).build().parseSignedClaims(token);
            return true;
        } catch (JwtException | IllegalArgumentException e) {
            return false;
        }
    }
    
    public String getUsername(String token) {
        return Jwts.parser().verifyWith(key).build()
            .parseSignedClaims(token).getPayload().getSubject();
    }
}

Complete REST API Security Configuration

@Configuration
@EnableWebSecurity
@EnableMethodSecurity
public class RestApiSecurityConfig {
    
    private final JwtTokenProvider jwtTokenProvider;
    
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        return http
            // Disable CSRF for stateless APIs
            .csrf(csrf -> csrf.disable())
            
            // Stateless session management
            .sessionManagement(session -> 
                session.sessionCreationPolicy(SessionCreationPolicy.STATELESS))
            
            // Exception handling
            .exceptionHandling(ex -> ex
                .authenticationEntryPoint((req, res, e) -> {
                    res.setContentType("application/json");
                    res.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
                    res.getWriter().write("{\"error\": \"Unauthorized\"}");
                })
                .accessDeniedHandler((req, res, e) -> {
                    res.setContentType("application/json");
                    res.setStatus(HttpServletResponse.SC_FORBIDDEN);
                    res.getWriter().write("{\"error\": \"Access Denied\"}");
                })
            )
            
            // Authorization rules
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/api/auth/**").permitAll()
                .requestMatchers("/api/public/**").permitAll()
                .requestMatchers("/actuator/health").permitAll()
                .requestMatchers(HttpMethod.GET, "/api/products/**").permitAll()
                .requestMatchers(HttpMethod.POST, "/api/products/**").hasRole("ADMIN")
                .requestMatchers("/api/admin/**").hasRole("ADMIN")
                .anyRequest().authenticated()
            )
            
            // Add JWT filter before UsernamePasswordAuthenticationFilter
            .addFilterBefore(
                new JwtAuthenticationFilter(jwtTokenProvider),
                UsernamePasswordAuthenticationFilter.class
            )
            
            // Security headers
            .headers(headers -> headers
                .contentSecurityPolicy(csp -> 
                    csp.policyDirectives("default-src 'self'"))
                .frameOptions(frame -> frame.deny())
            )
            
            // CORS configuration
            .cors(cors -> cors.configurationSource(corsConfigurationSource()))
            
            .build();
    }
    
    @Bean
    public CorsConfigurationSource corsConfigurationSource() {
        CorsConfiguration config = new CorsConfiguration();
        config.setAllowedOrigins(List.of("https://your-frontend.com"));
        config.setAllowedMethods(List.of("GET", "POST", "PUT", "DELETE", "OPTIONS"));
        config.setAllowedHeaders(List.of("Authorization", "Content-Type"));
        config.setExposedHeaders(List.of("X-Total-Count"));
        config.setAllowCredentials(true);
        config.setMaxAge(3600L);
        
        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
        source.registerCorsConfiguration("/api/**", config);
        return source;
    }
}

JWT Authentication Filter

public class JwtAuthenticationFilter extends OncePerRequestFilter {
    
    private final JwtTokenProvider tokenProvider;
    
    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                    HttpServletResponse response,
                                    FilterChain filterChain) 
            throws ServletException, IOException {
        
        String token = resolveToken(request);
        
        if (token != null && tokenProvider.validateToken(token)) {
            Authentication auth = tokenProvider.getAuthentication(token);
            SecurityContextHolder.getContext().setAuthentication(auth);
        }
        
        filterChain.doFilter(request, response);
    }
    
    private String resolveToken(HttpServletRequest request) {
        String bearerToken = request.getHeader("Authorization");
        if (bearerToken != null && bearerToken.startsWith("Bearer ")) {
            return bearerToken.substring(7);
        }
        return null;
    }
}

5. What is CSRF, and how does Spring Security protect against it?

Answer:

CSRF (Cross-Site Request Forgery) is an attack that tricks authenticated users into performing unintended actions on a web application by exploiting the trust the site has in the user's browser.

CSRF Attack Flow

+-------------+          +-------------+          +-------------+
|   Victim    |          |   Attacker  |          |   Bank.com  |
|   Browser   |          |   Website   |          |   Server    |
+-------------+          +-------------+          +-------------+
      |                        |                        |
      |  1. User logs into Bank.com                     |
      |------------------------------------------------>|
      |<-- Session cookie set --------------------------+
      |                        |                        |
      |  2. User visits attacker's site                 |
      |------------------>|                             |
      |                   |                             |
      |  3. Hidden form auto-submits:                   |
      |  <form action="bank.com/transfer">              |
      |     <input name="to" value="attacker">          |
      |     <input name="amount" value="10000">         |
      |  </form>                                        |
      |                   |                             |
      |  4. Browser sends request WITH session cookie   |
      |-------------------------------------------------->|
      |                        |                        |
      |  5. Bank processes transfer (victim is logged in)|
      |                        |                        |

Spring Security CSRF Protection

Spring Security generates a unique token per session that must be included in state-changing requests (POST, PUT, DELETE).

// CSRF token is automatically added to forms in Thymeleaf
<form th:action="@{/transfer}" method="post">
    <!-- Hidden field added automatically -->
    <input type="hidden" name="_csrf" th:value="${_csrf.token}"/>
    ...
</form>

// For AJAX requests (JavaScript)
const csrfToken = document.querySelector('meta[name="_csrf"]').content;
const csrfHeader = document.querySelector('meta[name="_csrf_header"]').content;

fetch('/api/transfer', {
    method: 'POST',
    headers: {
        [csrfHeader]: csrfToken,
        'Content-Type': 'application/json'
    },
    body: JSON.stringify(data)
});

When to Enable/Disable CSRF

Scenario CSRF Protection Reason
Traditional web app (server-rendered) ENABLE Uses session cookies, vulnerable to CSRF
Stateless REST API (JWT in header) DISABLE No cookies = no CSRF vulnerability
SPA with cookie-based JWT ENABLE Cookies are sent automatically

CSRF Configuration Examples

@Bean
public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
    return http
        // Option 1: Disable for stateless REST API
        .csrf(csrf -> csrf.disable())
        
        // Option 2: Enable with cookie-based token (for SPAs)
        .csrf(csrf -> csrf
            .csrfTokenRepository(CookieCsrfTokenRepository.withHttpOnlyFalse())
            .csrfTokenRequestHandler(new CsrfTokenRequestAttributeHandler())
        )
        
        // Option 3: Ignore specific endpoints
        .csrf(csrf -> csrf
            .ignoringRequestMatchers("/api/webhooks/**", "/api/public/**")
        )
        .build();
}

6. What is Spring Data JPA, and how does it simplify data access?

Answer:

Spring Data JPA is a library that significantly reduces boilerplate code for data access by providing repository abstractions over JPA/Hibernate. It enables developers to focus on business logic rather than data access implementation.

Architecture Overview

+-------------------+
|   Your Service    |
+-------------------+
         |
         v
+-------------------+
| Spring Data JPA   |
|   Repository      |  <-- You define interface only
+-------------------+
         |
         v
+-------------------+
| JPA/Hibernate     |  <-- Implementation auto-generated
+-------------------+
         |
         v
+-------------------+
|    DataSource     |
|   (Connection     |
|      Pool)        |
+-------------------+
         |
         v
+-------------------+
|    Database       |
+-------------------+

Key Features

Feature Description
Auto CRUD save(), findById(), findAll(), delete() methods generated
Query Methods findByName(), findByAgeGreaterThan() derived from method names
@Query Custom JPQL or native SQL queries
Pagination Built-in Pageable and Sort support
Auditing @CreatedDate, @LastModifiedBy automatic tracking
Specifications Dynamic queries with type-safe criteria

Entity Definition

@Entity
@Table(name = "products")
public class Product {
    
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    
    @Column(nullable = false, length = 100)
    private String name;
    
    @Column(precision = 10, scale = 2)
    private BigDecimal price;
    
    @Enumerated(EnumType.STRING)
    private Category category;
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "supplier_id")
    private Supplier supplier;
    
    @OneToMany(mappedBy = "product", cascade = CascadeType.ALL)
    private List<Review> reviews = new ArrayList<>();
    
    @CreatedDate
    private LocalDateTime createdAt;
    
    @LastModifiedDate
    private LocalDateTime updatedAt;
    
    // Getters, setters, constructors
}

Repository Example

public interface ProductRepository extends JpaRepository<Product, Long> {
    
    // Derived query methods - Spring generates implementation
    List<Product> findByCategory(Category category);
    List<Product> findByPriceBetween(BigDecimal min, BigDecimal max);
    Optional<Product> findByNameIgnoreCase(String name);
    List<Product> findByNameContainingOrderByPriceDesc(String keyword);
    
    // Custom JPQL query
    @Query("SELECT p FROM Product p WHERE p.price > :minPrice AND p.category = :category")
    List<Product> findExpensiveByCategory(@Param("minPrice") BigDecimal minPrice,
                                          @Param("category") Category category);
    
    // Native SQL query
    @Query(value = "SELECT * FROM products WHERE created_at > :date", nativeQuery = true)
    List<Product> findRecentProducts(@Param("date") LocalDateTime date);
    
    // Pagination support
    Page<Product> findByCategory(Category category, Pageable pageable);
    
    // Modifying queries
    @Modifying
    @Query("UPDATE Product p SET p.price = p.price * :multiplier WHERE p.category = :category")
    int updatePricesByCategory(@Param("multiplier") BigDecimal multiplier,
                               @Param("category") Category category);
}

7. What are the different types of repository interfaces in Spring Data?

Answer:

Repository Interface Hierarchy

                    +------------------+
                    |    Repository    |  <-- Marker interface
                    +------------------+
                            ^
                            |
                    +------------------+
                    | CrudRepository   |  <-- Basic CRUD operations
                    +------------------+
                            ^
                            |
               +------------+------------+
               |                         |
    +--------------------+    +------------------------+
    | ListCrudRepository |    | PagingAndSorting       |
    +--------------------+    | Repository              |  <-- + Pagination
                              +------------------------+
                                        ^
                                        |
                              +------------------------+
                              |    JpaRepository       |  <-- + JPA specific
                              +------------------------+

Repository Types Comparison

Interface Key Methods Use Case
Repository<T, ID> None (marker) Define custom repositories from scratch
CrudRepository<T, ID> save(), findById(), findAll(), delete(), count(), existsById() Basic CRUD without pagination
ListCrudRepository Same as Crud, returns List instead of Iterable When you need List return types
PagingAndSortingRepository findAll(Sort), findAll(Pageable) When you need pagination/sorting
JpaRepository<T, ID> flush(), saveAndFlush(), deleteInBatch(), getById() JPA apps needing full features (recommended)

Examples of Each Type

// 1. Basic marker - define only methods you need
public interface BasicUserRepository extends Repository<User, Long> {
    Optional<User> findById(Long id);
    User save(User user);
}

// 2. CrudRepository - standard CRUD
public interface UserCrudRepository extends CrudRepository<User, Long> {
    // Inherits: save, findById, findAll (Iterable), delete, count, existsById
}

// 3. PagingAndSorting - adds pagination
public interface UserPagingRepository extends PagingAndSortingRepository<User, Long> {
    // Additional: findAll(Sort), findAll(Pageable)
}

// 4. JpaRepository - full JPA features (most common choice)
public interface UserRepository extends JpaRepository<User, Long> {
    // Inherits all above PLUS:
    // - flush(), saveAndFlush()
    // - deleteAllInBatch(), deleteAllByIdInBatch()
    // - getReferenceById() (lazy proxy)
    // - findAll() returns List, not Iterable
    
    // Custom derived queries
    List<User> findByEmailContaining(String email);
    
    // With pagination
    Page<User> findByStatus(UserStatus status, Pageable pageable);
}

When to Use Which

  • JpaRepository: Default choice for 95% of use cases - full feature set
  • CrudRepository: Non-JPA stores (MongoDB, Redis) or minimal exposure
  • Repository: Maximum control - expose only specific methods

8. How do you implement caching in Spring Boot?

Answer:

Spring Boot provides a cache abstraction that works with various cache providers (Caffeine, Redis, Ehcache) through simple annotations.

Cache Flow

           +------------------+
           |   Method Call    |
           +------------------+
                    |
                    v
           +------------------+
           |  @Cacheable?     |
           +------------------+
                    |
          +---------+---------+
          |                   |
          v                   v
   +------------+      +-------------+
   | Cache HIT  |      | Cache MISS  |
   +------------+      +-------------+
          |                   |
          v                   v
   Return cached      Execute method
   value                     |
                             v
                      Store in cache
                             |
                             v
                      Return result

Cache Annotations

Annotation Purpose
@Cacheable Cache result; skip execution on cache hit
@CachePut Always execute method, update cache with result
@CacheEvict Remove entries from cache
@Caching Combine multiple cache operations
@CacheConfig Class-level cache defaults

Setup with Caffeine (In-Memory)

// build.gradle
implementation 'org.springframework.boot:spring-boot-starter-cache'
implementation 'com.github.ben-manes.caffeine:caffeine'

// application.yml
spring:
  cache:
    type: caffeine
    caffeine:
      spec: maximumSize=1000,expireAfterWrite=10m
@Configuration
@EnableCaching
public class CacheConfig {
    
    @Bean
    public CacheManager cacheManager() {
        CaffeineCacheManager manager = new CaffeineCacheManager();
        manager.setCaffeine(Caffeine.newBuilder()
            .maximumSize(1000)
            .expireAfterWrite(Duration.ofMinutes(10))
            .recordStats());  // Enable statistics
        return manager;
    }
}

Service Implementation

@Service
@CacheConfig(cacheNames = "products")  // Default cache name for class
public class ProductService {
    
    private final ProductRepository repository;
    
    // Cache by ID - returns cached value if exists
    @Cacheable(key = "#id")
    public Product findById(Long id) {
        log.info("Fetching from database: {}", id);  // Only logs on cache miss
        return repository.findById(id)
            .orElseThrow(() -> new NotFoundException("Product not found"));
    }
    
    // Conditional caching - only cache if price > 100
    @Cacheable(key = "#id", condition = "#result.price > 100")
    public Product findExpensiveById(Long id) {
        return repository.findById(id).orElseThrow();
    }
    
    // Update cache after modification
    @CachePut(key = "#product.id")
    public Product update(Product product) {
        return repository.save(product);
    }
    
    // Remove from cache when deleted
    @CacheEvict(key = "#id")
    public void delete(Long id) {
        repository.deleteById(id);
    }
    
    // Clear entire cache
    @CacheEvict(allEntries = true)
    public void clearCache() {
        log.info("Product cache cleared");
    }
    
    // Multiple cache operations
    @Caching(
        put = @CachePut(key = "#product.id"),
        evict = @CacheEvict(cacheNames = "productList", allEntries = true)
    )
    public Product save(Product product) {
        return repository.save(product);
    }
}

    @Bean
    public RedisCacheManager cacheManager(RedisConnectionFactory factory) {
        RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
            .entryTtl(Duration.ofMinutes(30))
            .serializeKeysWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new StringRedisSerializer()))
            .serializeValuesWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new GenericJackson2JsonRedisSerializer()))
            .disableCachingNullValues();
        
        return RedisCacheManager.builder(factory)
            .cacheDefaults(config)
            .withCacheConfiguration("products", 
                config.entryTtl(Duration.ofHours(1)))
            .withCacheConfiguration("users", 
                config.entryTtl(Duration.ofMinutes(15)))
            .build();
    }
}

9. Explain transaction management in Spring

Answer:

Spring provides declarative transaction management via @Transactional, abstracting the complexity of managing transactions across different data sources.

Transaction Flow

@Transactional method called
           |
           v
+----------------------+
| TransactionManager   |
| begins transaction   |
+----------------------+
           |
           v
+----------------------+
| Execute method body  |
|   - DB operations    |
|   - Business logic   |
+----------------------+
           |
    +------+------+
    |             |
    v             v
SUCCESS        EXCEPTION
    |             |
    v             v
+--------+   +-----------+
| COMMIT |   | ROLLBACK  |
+--------+   +-----------+

Transaction Propagation Types

Propagation Behavior
REQUIRED (default) Join existing TX, or create new if none exists
REQUIRES_NEW Always create new TX, suspend existing
MANDATORY Must run within existing TX, throw exception if none
SUPPORTS Use TX if exists, run non-transactional if not
NOT_SUPPORTED Suspend existing TX, run non-transactional
NEVER Throw exception if TX exists
NESTED Create savepoint within existing TX

Isolation Levels

Isolation Dirty Read Non-Repeatable Phantom
READ_UNCOMMITTED Possible Possible Possible
READ_COMMITTED Prevented Possible Possible
REPEATABLE_READ Prevented Prevented Possible
SERIALIZABLE Prevented Prevented Prevented

@Transactional Examples

@Service
public class OrderService {
    
    private final OrderRepository orderRepository;
    private final PaymentService paymentService;
    private final NotificationService notificationService;
    
    // Basic transaction - rollback on any RuntimeException
    @Transactional
    public Order createOrder(OrderRequest request) {
        Order order = new Order(request);
        order = orderRepository.save(order);
        paymentService.processPayment(order);  // Same transaction
        return order;
    }
    
    // Custom rollback rules
    @Transactional(
        rollbackFor = Exception.class,           // Rollback on checked exceptions
        noRollbackFor = NotificationException.class  // Don't rollback for this
    )
    public Order createOrderWithNotification(OrderRequest request) {
        Order order = orderRepository.save(new Order(request));
        notificationService.sendConfirmation(order);  // May throw NotificationException
        return order;
    }
    
    // Read-only transaction (optimization hint)
    @Transactional(readOnly = true)
    public List<Order> findAllOrders() {
        return orderRepository.findAll();
    }
    
    // Separate transaction for audit logging
    @Transactional(propagation = Propagation.REQUIRES_NEW)
    public void logAudit(String action, Long orderId) {
        // This commits independently, even if outer TX rolls back
        auditRepository.save(new AuditLog(action, orderId));
    }
    
    // Timeout handling
    @Transactional(timeout = 30)  // 30 seconds
    public void processLargeOrder(OrderRequest request) {
        // Long-running operation
    }
}

Common Pitfalls

  • Self-invocation: Calling @Transactional method from same class bypasses proxy
  • Checked exceptions: By default, only RuntimeException triggers rollback
  • Private methods: @Transactional doesn't work on private methods
  • Final classes/methods: Cannot be proxied (use interface or CGLIB)
// WRONG - self-invocation bypasses @Transactional
@Service
public class OrderService {
    public void process() {
        createOrder();  // @Transactional NOT applied!
    }
    
    @Transactional
    public void createOrder() { ... }
}

// CORRECT - inject self or use separate service
@Service
public class OrderService {
    @Autowired
    private OrderService self;  // Inject proxy
    
    public void process() {
        self.createOrder();  // Goes through proxy
    }
}

10. What is the difference between Lazy and Eager loading in JPA?

Answer:

Loading Strategies Comparison

Aspect LAZY EAGER
When loaded On first access Immediately with parent
Initial query Lighter (parent only) Heavier (with JOINs)
N+1 risk Yes, if accessed in loop No (loaded upfront)
Memory usage Lower initially Higher (all data loaded)
Default for @OneToMany, @ManyToMany @ManyToOne, @OneToOne

Visual Comparison

LAZY Loading:
                                                      
  findById(1)          Access orders              Additional Query
       |                    |                           |
       v                    v                           v
+------------+      +----------------+         +----------------+
| SELECT     |      | user.getOrders(|  ----> | SELECT * FROM  |
| * FROM     |      |                |         | orders WHERE   |
| users      |      +----------------+         | user_id = 1    |
| WHERE id=1 |                                 +----------------+
+------------+


EAGER Loading:

  findById(1)
       |
       v
+-------------------------------------------+
| SELECT u.*, o.*                           |
| FROM users u                              |
| LEFT JOIN orders o ON u.id = o.user_id    |
| WHERE u.id = 1                            |
+-------------------------------------------+
       |
       v
  All data loaded immediately

Entity Configuration

@Entity
public class User {
    @Id
    private Long id;
    
    private String name;
    
    // LAZY: Orders loaded only when accessed
    @OneToMany(mappedBy = "user", fetch = FetchType.LAZY)
    private List<Order> orders;
    
    // EAGER: Profile always loaded with user
    @OneToOne(fetch = FetchType.EAGER)
    private UserProfile profile;
}

@Entity  
public class Order {
    @Id
    private Long id;
    
    // EAGER (default for @ManyToOne): User loaded with order
    @ManyToOne(fetch = FetchType.LAZY)  // Override to LAZY
    private User user;
}

N+1 Problem and Solutions

// N+1 PROBLEM - 1 query for users + N queries for orders
List<User> users = userRepository.findAll();  // 1 query
for (User user : users) {
    user.getOrders().size();  // N additional queries!
}

// SOLUTION 1: JOIN FETCH in JPQL
@Query("SELECT u FROM User u JOIN FETCH u.orders")
List<User> findAllWithOrders();

// SOLUTION 2: @EntityGraph
@EntityGraph(attributePaths = {"orders", "orders.items"})
List<User> findAll();

// SOLUTION 3: @BatchSize (fetch in batches)
@OneToMany(mappedBy = "user")
@BatchSize(size = 20)  // Load orders for 20 users at once
private List<Order> orders;

// SOLUTION 4: DTO projection (best for read-only)
@Query("SELECT new com.example.UserOrderDTO(u.name, COUNT(o)) " +
       "FROM User u LEFT JOIN u.orders o GROUP BY u.id")
List<UserOrderDTO> findUserOrderCounts();

Best Practices

  • Default to LAZY for collections (@OneToMany, @ManyToMany)
  • Consider LAZY for @ManyToOne if parent not always needed
  • Use JOIN FETCH or @EntityGraph when you need related data
  • Use DTO projections for read-only queries
  • Enable SQL logging in dev to detect N+1 issues
# Detect N+1 in development
spring:
  jpa:
    show-sql: true
    properties:
      hibernate:
        format_sql: true
logging:
  level:
    org.hibernate.stat: DEBUG
    org.hibernate.SQL: DEBUG
